use BatallaDeRaces;
CREATE TABLE weapons (
	WEAPON_ID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	WEAPON_NAME varchar(50),
	WEAPON_SPEED int,
	WEAPON_STRENGHT int,
	WEAPON_IMAGE_PATH varchar(100),
    WEAPON_POINTS int
);

CREATE TABLE warriors (
	WARRIOR_ID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	WARRIOR_NAME varchar(50),
	WARRIOR_HEALTH int,
	WARRIOR_STRENGHT int,
	WARRIOR_SPEED int,
	WARRIOR_AGILITY int,
	WARRIOR_DEFENSE int,
	WARRIOR_IMAGE_PATH varchar(100),
	WARRIOR_POINTS int    	
);

CREATE TABLE players ( 
	PLAYER_ID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
	PLAYER_NAME varchar(50),
	SCORE int,
	ENEMIES_SLAYED int
);

CREATE TABLE weapons_available (
	WARRIOR_ID INT NOT NULL,
	WEAPON_ID int not null,
	FOREIGN KEY (WARRIOR_ID) REFERENCES warriors (WARRIOR_ID),
    	FOREIGN KEY (WEAPON_ID) REFERENCES weapons (WEAPON_ID)
);

CREATE TABLE battle (
	BATTLE_ID INT NOT NULL PRIMARY KEY auto_increment,
	PLAYER_ID INT,
	WARRIOR_ID INT NOT NULL,
	WARRIOR_WEAPON_ID INT NOT NULL,
	OPPONENT_ID INT NOT NULL,
	OPPONENT_WEAPON_ID INT NOT NULL,
	INJURIES_CAUSED INT,
	INJURIES_SUFFERED INT,
	BATTLE_POINTS INT,
	FOREIGN KEY (WARRIOR_ID) REFERENCES warriors (WARRIOR_ID),
	FOREIGN KEY (WARRIOR_WEAPON_ID) REFERENCES weapons (WEAPON_ID),
	FOREIGN KEY (PLAYER_ID) REFERENCES players (PLAYER_ID)
);
